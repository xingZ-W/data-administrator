```csharp
class 极耳焊接 : Father
{

    private RecipeManager _recipeManager;
    
    public 极耳焊接(Form1 form, string FilePath2)

        this.form = form;
        FilePath = FilePath2;
        MES_operator = "user";
        mesClass = new MESClass(form);
        
        // 【新增】启动高频采集器

        // 订阅焊接完成事件
        _collector.OnWeldingCompleted += OnWeldingCompleted;
        
        Thread 心跳 = new Thread(xintiao);
        心跳.IsBackground = true;
        心跳.Start();
        
        MySignal signal = new MySignal(this.form);
        signal.AddListSignal(new Signal(FindIniPath("进站 1 触发")[0], 工位 1 进站，工位 1 进站复位));
        signal.AddListSignal(new Signal(FindIniPath("出站 1 触发")[0], 工位 1 出站，工位 1 出站复位));
    }
    
    // 【新增】焊接完成回调
    private async void OnWeldingCompleted(List<WeldingSample> data)
    {
        // TODO: 实现 AI 质量判定
        var judgeResult = await JudgeQuality(data);
        
        form.Invoke(new Action(() => 
        {
            if (judgeResult.IsPass)
            {
                form.P_Class.TxtLog("质量判定：OK", 2);
            }
            else
            {
                form.P_Class.TxtLog($"质量判定：NG - {judgeResult.DefectType}", 1);
                // 触发 NG 信号
                form.PLC_ui.write(FindIniPath("出站 1NG")[0], true);
            }
        }));
    }
    
    // 修改进站方法，增加自动配方调用
    public override string 工位 1 进站(string Code = "")
    {
        StationInTime = DateTime.Now.ToString("yyyy 年 MM 月 dd 日 HH:mm:ss");
        form.P_Class.TxtLog("接收极耳焊接进站 1 信号，开始进站！", 2);
        moduleCode1 = form.PLC_ui.Read_String(FindIniPath("WIP 码")[0]);
        form.P_Class.TxtLog("极耳焊接 1 进站 WIP 码：" + moduleCode1, 2);
        
        if (moduleCode1 == " ")
        {
            // ... 原有空码处理逻辑 ...
            return "";
        }
        
        // 【新增】自动加载配方
        if (!form.采集模式.Text.Contains("离线"))
        {
            var task = Task.Run(async () => 
            {
                var result = await _recipeManager.AutoLoadRecipe(moduleCode1);
                if (!result.IsSuccess)
                {

                    {

                        DataGridViewClass.AddRows(form.dataGridView1,

            task.Wait(); // 等待配方加载完成
```

### PLC 地址配置

## ❓ 常见问题

### Q1: 高频采集器启动失败？
**A**: 检查 PLC 通讯是否正常，确认 PLC 地址配置正确。

### Q2: 配方加载超时？
**A**: 检查网络连接，确认 MES 服务器地址正确。如果网络异常，系统会自动使用本地缓存。

### Q3: 数据库文件找不到？
**A**: 确保在 `Program.cs` 或 `Form1_Load` 中调用了数据库初始化代码。

### Q4: NuGet 包安装失败？
**A**: 检查项目目标框架是否为 `.NET Framework 4.7.2` 或更高版本。
